cmake_minimum_required(VERSION 3.10)

# 1. Name your project
project(NvEncodeTest)

# 2. Set the C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include(FetchContent)
FetchContent_Declare(
    datachannel
    GIT_REPOSITORY https://github.com/paullouisageneau/libdatachannel.git
    GIT_TAG 082f2da
)
FetchContent_MakeAvailable(datachannel)

add_executable(NvEncodeTest test.cpp)

target_include_directories(NvEncodeTest PRIVATE "${CMAKE_SOURCE_DIR}/include")

target_link_libraries(NvEncodeTest PRIVATE 
    "${CMAKE_SOURCE_DIR}/lib/nvEncodeAPI.lib"
    d3d11
    datachannel
)

add_custom_command(TARGET NvEncodeTest POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    $<TARGET_FILE:datachannel>
    $<TARGET_FILE_DIR:NvEncodeTest>
)
